准备工作

步骤

  1. 配置settings.py 文件

broker这里配置的redis

backend任务结果也是存到redis中

格式为Json

#Celery配置
CELERY_BROKER_URL = 'redis://127.0.0.1:6379/0'
CELERY_RESULT_BACKEND = 'redis://127.0.0.1:6379/1'
CELERY_RESULT_SERIALIZER = 'json'
  1. 在根目录的init.py文件配置
from __future__ import absolute_import, unicode_literals
from .celery import app as celery_app

__all__ = ['celery_app',]
  1. 根目录创建celery.py
from __future__ import absolute_import, unicode_literals
import os
from celery import Celery

# 设置默认 DJANGO_SETTINGS_MODULEcelery. 命令行程序的环境变量
os.environ.setdefault('DJANGO_SETTINGS_MODULE', 'DEMO.settings') # DEMO.settings 配置的是自己的项目



# 实例化Celery对象
app = Celery('DEMO')

# 绑定配置文件
app.config_from_object('django.conf:settings', namespace='CELERY')


# 自动发现APPS.tasks.py模块
app.autodiscover_tasks()
  1. 在app/tasks.py

这是一个发送邮件的例子

from __future__ import absolute_import, unicode_literals


from django.conf import settings
from django.core.mail import send_mail

from celery import Celery, shared_task
from celery.task import task

# 任务处理
@shared_task
def send_mail_task():
    send_mail(
        subject='Hi.',
        message='To, Tao',
        from_email=settings.EMAIL_HOST_USER,
        recipient_list=['569781231@qq.com', ],
        fail_silently=False
    )
  1. 视图views.py
from django.http import HttpResponse
from django.views import View

from .tasks import send_mail_task

class IndexView(View):
    def get(self, request):
        send_mail_task.delay()
        return HttpResponse('Get Ok')

周期任务

编辑根目录下的settings.py

from celery.schedules import crontab
CELERY_BEAT_SCHEDULE = {
    # 周期性任务
    'task-one': {
        'task': 'front.tasks.print_task',	
        'schedule': 5.0, # 每5秒执行一次  # crontab(day_of_week=1,hour=7) 每个星期一的上午7点执行
        # 'args': ()
    }
}

APP下的tasks.py

from celery.task import task

@task
def print_task():
    print('Hello. task')
    return 'Hello'

如果使用定时任务需要添加beat . 且单独启动beat服务

$ celery -A DEMO beat -l info
