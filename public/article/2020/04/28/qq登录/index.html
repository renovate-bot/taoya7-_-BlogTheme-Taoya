from urllib.parse import urlencode
import  requests
import json

class OAuthQQ:

    def __init__(self, client_id, client_key, redirect_uri):
        self.client_id = client_id
        self.client_key = client_key
        self.redirect_uri = redirect_uri

    def get_auth_url(self):
        '''
        获取授权页面的网址
        '''
        params = {
            'which': 'Login',
            'display': 'pc',
            'response_type': 'code',
            'client_id': self.client_id,
            'redirect_uri': self.redirect_uri,
            'scope': 'get_user_info',
            'state': 1
        }
        url = 'https://graph.qq.com/oauth2.0/show?%s'%urlencode(params)
        return url

def get_access_token(self, code):
    '''
    	根据Code获取access_token
    '''
    params = {
        'grant_type': 'authorization_code',
        'client_id': self.client_id,
        'client_secret': self.client_key,
        'code': code,
        'redirect_uri': self.redirect_uri
    }
    url = 'https://graph.qq.com/oauth2.0/token?%s'%urlencode(params)

    response = requests.get(url).text
    access_token = response.split('=')[1].split('&')[0]
    self.access_token = access_token
    return access_token

获取授权ID

def get_open_id(self):
    '''获取QQ的OpenID'''
    params = {
        'access_token': self.access_token
    }
    url = 'https://graph.qq.com/oauth2.0/me'
    response = requests.get(url, params=params).text
    r_str = str(response)[9:-3] # 去掉callback
    r_json = json.loads(r_str)

    openid = r_json['openid']
    self.openid = openid
    return openid

def get_qq_info(self):
    '''获取用户资料信息'''
    params = {
        'access_token': self.access_token,
        'oauth_consumer_key': self.client_id,
        'openid': self.openid
    }

    url = 'https://graph.qq.com/user/get_user_info'
    response = requests.get(url, params=params).text

    info = json.loads(response)
    return info

from urllib.parse import urlencode
import  requests
import json

class OAuthQQ:

    def __init__(self, client_id, client_key, redirect_uri):
        self.client_id = client_id
        self.client_key = client_key
        self.redirect_uri = redirect_uri

    def get_auth_url(self):
        '''
        获取授权页面的网址
        '''
        params = {
            'which': 'Login',
            'display': 'pc',
            'response_type': 'code',
            'client_id': self.client_id,
            'redirect_uri': self.redirect_uri,
            'scope': 'get_user_info',
            'state': 1
        }
        url = 'https://graph.qq.com/oauth2.0/show?%s'%urlencode(params)
        return url

    def get_access_token(self, code):
        '''
        根据Code获取access_token
        '''
        params = {
            'grant_type': 'authorization_code',
            'client_id': self.client_id,
            'client_secret': self.client_key,
            'code': code,
            'redirect_uri': self.redirect_uri
        }
        url = 'https://graph.qq.com/oauth2.0/token?%s'%urlencode(params)

        response = requests.get(url).text
        # access_token=851BA27D849C1B4F913F6024C373BA17&expires_in=7776000&refresh_token=873139F27AB656AAE3D40A84E534A80C
        access_token = response.split('=')[1].split('&')[0]
        self.access_token = access_token
        return access_token

    def get_open_id(self):
        '''获取QQ的OpenID'''

            'access_token': self.access_token
